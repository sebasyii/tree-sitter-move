module 0x1::MatchLambda {
    fun t() {
        match (x) {
            1 => 2,
            y => y + 1,
            Option::Some(z) => z,
            Option::None => 0
        };

        match (x) {
            1 => 2,
            y if y > 10 => y,
        };

        let f = |x| x + 1;
        let g = |x: u64, y| x + y;
        let h = || 1;
        let m = move |x| x;
    }
}

module 0x1::Structs {
    struct Empty;

    struct Native;

    native struct NativeStruct;

    struct WithAbilities has copy, drop;

    struct WithFields has key {
        f: u64,
        g: bool
    }

    struct Positional(u64, bool) has copy;

    struct Generic<T: drop> has store {
        x: T
    }

    struct MultiAbility<T: copy + drop + store> has key, store {
        a: T
    }

    struct Phantom<phantom T> {
        a: u64
    }

    struct MultiGeneric<T, U> {
        x: T,
        y: U
    }

    struct MultiGenericConstraints<T: copy + drop, U: key> {
        x: T,
        y: U
    }

    enum Color {
        Red,
        Green,
        Blue
    }

    enum Shape {
        Circle(u64),
        Rectangle { width: u64, height: u64 }
    }

    enum GenericEnum<T> has drop {
        None,
        Some(T)
    }

    enum MultiGenericEnum<T, U> {
        One(T),
        Two(U)
    }

    enum MultiGenericEnumConstraints<T: copy + drop, U: key> {
        One(T),
        Two(U)
    }
}
